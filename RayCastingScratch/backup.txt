	public void update(KeyHandler keyH) {
		if (keyH.fovIncrease && ray_count<180) {
			ray_count += 1;
		} if (keyH.fovDecrease && ray_count>1) {
			ray_count -= 1;
		}
	}
	public void draw(Graphics2D g2, KeyHandler keyH) {
		
		
		Line2D lines[] = math();
		double lineH;

		int lineO; 
		double ca;
		double defaultWidth = 8.5;
		double Width = defaultWidth/(ray_count/60.0);
		
		double ty, tx; //texture xy
		double ty_step, ty_offset;
		int r, g, b, c;
		double shade;
		
		int line_num = 0;
		for (Line2D line : lines) {
			
			if (!keyH.fisheye) {
				ca = DR*(ray_count/2)-DR*line_num;
				
				if (ca<0) {
					ca += 2*PI;
				} else if (ca > 2*PI) {
					ca -= 2*PI;
				}
				ca = Math.cos(ca);
			} else {
				ca = 1;
			}
			
			
			if (line.getX1()== -999) {
				line = new Line2D.Double(gp.player.x, line.getY1(), line.getX2(), line.getY2());
				shade = 0.5;
			} else {
				shade = 1;
			}
			
			g2.setStroke(new BasicStroke(1));
			g2.setColor(new Color(100, 100, 100));
			g2.draw(line);
			
			lineH = (gp.mapSize*320/(length(line)*ca));
			ty_step = 32.0/lineH;
			ty_offset = 0; 
			if (lineH>320) {
				ty_offset = (lineH-320)/2.0;
				lineH = 320;
			}
			lineO = (int) (160-lineH/2);
			ty = ty_offset*ty_step;
			//tx = (int) (/2.0)%32;
			
			
			g2.setStroke(new BasicStroke((float) (Width)));
			//g2.draw(new Line2D.Double((double) (line_num*Width)+512+(Width/2), lineO, (double) (line_num*Width)+512+(Width/2), lineH+lineO));
			

			for (int y=0;y<lineH;y++) {
				c = All_Textures[(int) ty*32];
				r = 0;
				g = (int) (c*50*shade);
				b = (int) (c*50*shade);
				g2.setColor(new Color(r, g, b));
				g2.draw(new Line2D.Double((double) (line_num*Width)+512+(Width/2), y+lineO, (double) (line_num*Width)+512+(Width/2), y+lineO));
				ty+=ty_step;
			}
		
			line_num++;
		}
	}
	
	public Line2D[] math() {
		Line2D[] rays = new Line2D[ray_count];
		
		px = (int) gp.player.x;
		py = (int) gp.player.y;
		
		Line2D vertical = null;
		Line2D horizontal = null;
		
		ra = gp.player.a-DR*(ray_count/2);
		
		if (ra<0) {
			ra += 2*PI;
		} else if (ra > 2*PI) {
			ra -= 2*PI;
		}
		for (r=0;r<ray_count;r++) {
			
			// HORIZONTAL LINES 
			dof = 0;
			double aTan = -1/Math.tan(ra);
			
			if(ra>PI) {
				ry = (((int)py>>6)<<6)-0.0001;
				rx = (py-ry)*aTan+px;
				yo = -64;
				xo = -yo*aTan;
			} else if (ra<PI) {
				ry = (((int)py>>6)<<6) + 64;
				rx = (py-ry)*aTan+px;
				yo = 64;
				xo = -yo*aTan;
			} else if (ra==0 || ra==PI) {
				rx=px;
				ry=py;
				dof=8;
			}
			while (dof<8) {
				mx = (int) rx>>6;
				my = (int) ry>>6;
				
				if (mx>=0&&mx<8&&my>=0&&my<8&&gp.map.map[my][mx] == 1) {
					dof=8;
				} else {
					rx += xo;
					ry += yo;
					dof += 1;
				}
			}
			
			horizontal = new Line2D.Double(px, py, rx, ry);
		
		
			// VERTICAL LINES
			dof = 0;
			double nTan = -Math.tan(ra);
			
			if(ra>PI2 && ra<PI3) {
				rx = (((int)px>>6)<<6)-0.0001;
				ry = (px-rx)*nTan+py;
				xo = -64;
				yo = -xo*nTan;
			} else if (ra<PI2 || ra>PI3) {
				rx = (((int)px>>6)<<6) + 64;
				ry = (px-rx)*nTan+py;
				xo = 64;
				yo = -xo*nTan;
			} else if (ra==0 || ra==PI) {
				rx=px;
				ry=py;
				dof=8;
			}
			while (dof<8) {
				mx = (int) rx>>6;
				my = (int) ry>>6;
				
				if (mx>=0&&mx<8&&my>=0&&my<8&&gp.map.map[my][mx] == 1) {
					dof=8;
				} else {
					rx += xo;
					ry += yo;
					dof += 1;
				}
			}
			
			if (ra<0) {
				ra += 2*PI;
			} else if (ra > 2*PI) {
				ra -= 2*PI;
			}
			
			vertical = new Line2D.Double(px, py, rx, ry);
			
			if (Math.min(length(vertical), length(horizontal)) == length(vertical)) {
				rays[r] = new Line2D.Double(-999, vertical.getY1(), vertical.getX2(), vertical.getY2());
			} else {
				rays[r] = horizontal;
			}
			
			ra += DR;
			if (ra<0) {
				ra += 2*PI;
			} else if (ra > 2*PI) {
				ra -= 2*PI;
			}
			
		}

		return rays;
		
	}
	
	public double length(Line2D line) {
		return (Math.sqrt(Math.pow((line.getX1()-line.getX2()), 2) + Math.pow((line.getY1()-line.getY2()), 2)));
	}
